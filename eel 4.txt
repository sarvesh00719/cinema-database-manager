#include <stdio.h>
#include <string.h>

struct Cinema {
    char customer_name[20];
    char movie_name[20];
    char show_time[15];
    int seats;
    int ticket_price;
    int date;    // DDMMYYYY format
};

int main() {
    struct Cinema booking[10], temp;
    int n;

    printf("Enter number of bookings (max 10): ");
    if (scanf("%d", &n) != 1 || n < 1 || n > 10) {
        printf("Invalid number of bookings. Exiting.\n");
        return 1;
    }

    for (int i = 0; i < n; i++) {
        printf("\n--- ENTER DETAILS FOR BOOKING %d ---\n", i + 1);

        printf("ENTER CUSTOMER NAME: ");
        scanf("%s", booking[i].customer_name);

        printf("ENTER MOVIE NAME: ");
        scanf("%s", booking[i].movie_name);

        printf("ENTER SHOW TIME: ");
        scanf("%s", booking[i].show_time);

        printf("ENTER NUMBER OF SEATS: ");
        scanf("%d", &booking[i].seats);

        printf("ENTER TOTAL TICKET PRICE: ");
        scanf("%d", &booking[i].ticket_price);

        printf("ENTER SHOW DATE (DDMMYYYY): ");
        scanf("%d", &booking[i].date);
    }

    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (booking[j].ticket_price > booking[j + 1].ticket_price) {
                temp = booking[j];
                booking[j] = booking[j + 1];
                booking[j + 1] = temp;
            }
        }
    }

    printf("\n\n================ CINEMA BOOKING DETAILS ================\n");

    // Header with alignment
    printf("%-15s %-15s %-12s %-8s %-10s %-12s\n",
           "NAME", "MOVIE", "TIME", "SEATS", "PRICE", "DATE");

    printf("--------------------------------------------------------------------------\n");

    // Table Rows aligned
    for (int i = 0; i < n; i++) {
        printf("%-15s %-15s %-12s %-8d %-10d %-12d\n",
               booking[i].customer_name,
               booking[i].movie_name,
               booking[i].show_time,
               booking[i].seats,
               booking[i].ticket_price,
               booking[i].date);
    }

    printf("====================================================================");
    return 0;
}
